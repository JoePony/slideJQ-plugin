    /* 轮播 */
$('slideWrap1').slideJQ({        //用容器元素调用slideJQ组件，必选项
    slideList: '.slide-list',        //幻灯列表元素，必选项
    navList: {
        ele: '.index-list',       //索引按钮列表元素，可选项
        activeClass: 'active'     //高亮样式，可选项
    },
    titleList: '.slide-title-list',    //标题列表元素，可选项
    arrowPrev: '.prev',       //前一张按钮，可选项
    arrowNext: '.next',       //下一张按钮，可选项
    speed: 280,               //播放速度，可选项，默认为300毫秒
    animate: {
        type: 'slide'  //类型可选slide/fade/blur/scale/rotate/default，可选项，默认slide(支持移动端touch)
        //className:''     //fade、blur、scale、rotate配合实现效果的相应css3样式名
    },
    aniamtionType: 'swing',    //动画类型，仅slide支持，可选项，默认swing
    autoPlay: true,       //是否自动播放，可选项，默认自动播放
    duration: 1500,       //持续时间，可选项，默认2500毫秒
    initialIndex: 2,      //指定最初高亮索引，可选项，默认为1
    mouseType: 'click mouseover',    //索引button上的鼠标触发事件，可选项，默认为click mouseover
    titleHidden:false,   // 标题是否隐藏，可选项，默认为false
    navHidden:false,      // 索引按钮是否隐藏，可选项，默认为false
    remainNav:false //默认根据图片数量自动生成索引按钮，并删除原有静态索引元素。要保持原有元素则选true
});
